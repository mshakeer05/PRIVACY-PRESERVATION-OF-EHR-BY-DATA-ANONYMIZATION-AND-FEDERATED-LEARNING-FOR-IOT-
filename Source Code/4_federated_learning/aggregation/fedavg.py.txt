import numpy as np

def fed_avg(model_weights_list):
    """Perform Federated Averaging over list of model weight dicts."""
    avg_weights = {}
    keys = model_weights_list[0].keys()

    for key in keys:
        avg_weights[key] = np.mean([client_weights[key] for client_weights in model_weights_list], axis=0)
    
    print("[FedAvg] Aggregation complete.")
    return avg_weights
